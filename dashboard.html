<script src="https://www.gstatic.com/firebasejs/9.22.1/firebase-app-compat.js"></script>

<script src="https://www.gstatic.com/firebasejs/9.22.1/firebase-database-compat.js"></script>

<script>

const firebaseConfig = { databaseURL: "TU_WKLEJ_SWOJ_URL_Z_FIREBASE" };

firebase.initializeApp(firebaseConfig);

const db = firebase.database();


db.ref('wyniki').on('value', (snapshot) => {

    const data = snapshot.val();

    if (!data) return;

    const arr = Object.values(data);

    const n = arr.length;

    

    document.getElementById('count').innerText = n;

    document.getElementById('avg-flex').innerText = Math.round(arr.reduce((s,r)=>s+r.flex,0)/n) + '%';

    document.getElementById('avg-myths').innerText = (arr.reduce((s,r)=>s+r.myths,0)/n).toFixed(1);

    document.getElementById('status').innerText = "✓ Aktywny | Ostatni wynik: " + new Date().toLocaleTimeString();


    let genHtml = '';

    const gens = [{k:'BB', n:'Baby Boomers'}, {k:'X', n:'Gen X'}, {k:'Y', n:'Millenialsi'}, {k:'Z', n:'Gen Z'}];

    gens.forEach(g => {

        const avg = (arr.reduce((s,r)=>s+r[g.k],0)/n).toFixed(1);

        const pct = (avg / 7.5) * 100;

        genHtml += `<div class="bar-row"><div class="bar-lbl">${g.n} (${avg} pkt)</div><div class="bar-wrap"><div class="bar-fill" style="width:${pct}%"></div></div></div>`;

    });

    document.getElementById('gen-stats').innerHTML = genHtml;

});

</script>
